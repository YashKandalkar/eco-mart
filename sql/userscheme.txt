create table users(
   id SERIAL NOT NULL,
   emailid VARCHAR(25) NOT NULL primary key  ,
   firstname varchar(20),
   lastname varchar(20),
   password varCHAR(40) NOT NULL,
   contact_no varchar(15) not null,
   category varchar(25) not null,
   address varchar(70) not null
   points int not null default 50
);

create table products(
	product_id  SERIAL  primary key,
	seller_emailid varchar(25),
	image_path varchar(200),
	product_name varchar(25),
	product_category varchar(25),
	description Text,
	price int,
	quantity int,
   points int not null default 0,
	FOREIGN KEY (seller_emailid) REFERENCES users(emailid)
	
);

CREATE TABLE cart (
    id SERIAL NOT NULL,
	emailid varchar(25),
    product_id int,
	quantity int NOT NULL DEFAULT 1,
	price int,
	status boolean not null DEFAULT FALSE,
	PRIMARY KEY (id)
);

create table orders(
	order_id  SERIAL  primary key,
	customer_emailid varchar(25),
	product_id int,
	quantity int,
	price int,
	order_date date not NULL DEFAULT CURRENT_DATE ,
	FOREIGN KEY (customer_emailid) REFERENCES users(emailid),
	FOREIGN KEY (product_id) REFERENCES products(product_id)
);

create table blogs(
	id  SERIAL  primary key,
	admin_emailid varchar(25),
	title varchar(25),
	description Text,
	subtitle varchar(25),
	sub_description Text,
	thumbnail varchar(300),
	date date not NULL DEFAULT CURRENT_DATE,
	FOREIGN KEY (admin_emailid) REFERENCES users(emailid)
)
