create table users(
   id INT NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1 INCREMENT BY 1),
   emailid VARCHAR(25) NOT NULL primary key enforced ,
   firstname varchar(20),
   lastname varchar(20),
   password VARCHAR(40) NOT NULL,
   contact_no varchar(15) not null,
   category varchar(25) not null,
   address varchar(70) not null
  
);


create table products(
	product_id INT NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1 INCREMENT BY 1), 
	user_emailid varchar(25),
	product_name varchar(25),
	product_category varchar(25),
	description varchar(255),
	price int,
	quantity int,
	FOREIGN KEY (user_emailid) REFERENCES users(emailid)
);
